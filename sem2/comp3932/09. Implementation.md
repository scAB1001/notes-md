```csharp
        private int activeHandId = -1;
        private float trackingLossTimer = 0f;

        private Vector3 positionOffset;
        private Quaternion rotationOffset;

        private void Start()
        {
            HandTrackingHintManager.RequestHandTrackingHints(new string[] { "high_hand_fidelity", "microgestures" });
            if (statusText != null) { statusText.text = "SYSTEM IDLE"; statusText.color = Color.white; }
        }

        private void Update()
        {
            if (leapProvider.CurrentFrame == null) return;
            Frame frame = leapProvider.CurrentFrame;

            if (!IsGrabbing)
                ScanForStrictPinch(frame);
            else
                ProcessKinematicTransformation(frame);
        }

        private void ScanForStrictPinch(Frame frame)
        {
            foreach (Hand hand in frame.Hands)
            {
                Vector3 thumbTip = hand.Fingers[(int)Finger.FingerType.THUMB].TipPosition;
                Vector3 indexTip = hand.Fingers[(int)Finger.FingerType.INDEX].TipPosition;
                float actualPinchDist = Vector3.Distance(thumbTip, indexTip);

                // STRICT CHECK: Fingers must be close, AND hand must not be in a fist
                if (actualPinchDist <= pinchEnterThreshold && hand.GrabStrength <= maxAllowedGrabStrength)
                {
                    InitiateKinematicLink(hand, thumbTip, indexTip);
                    break;
                }
            }
        }

        private void InitiateKinematicLink(Hand hand, Vector3 thumbTip, Vector3 indexTip)
        {
            IsGrabbing = true;
            activeHandId = hand.Id;
            trackingLossTimer = 0f;
            metricsLogger.RegisterGrabEvent();

            if (statusText != null) { statusText.text = "GRABBING"; statusText.color = Color.green; }

            Vector3 pinchCenter = Vector3.Lerp(thumbTip, indexTip, 0.5f);
            positionOffset = solidLiver.position - pinchCenter;
            rotationOffset = Quaternion.Inverse(hand.Rotation) * solidLiver.rotation;
        }

        private void ProcessKinematicTransformation(Frame frame)
        {
            Hand activeHand = frame.Hand(activeHandId);

            // OCCLUSION GRACE PERIOD: If hand is temporarily lost
            if (activeHand == null)
            {
                trackingLossTimer += Time.deltaTime;
                if (trackingLossTimer >= occlusionGracePeriod)
                {
                    DropLiver();
                }
                return; // Maintain position while in grace period
            }

            trackingLossTimer = 0f; // Reset timer if hand is found

            // RELEASE CHECK
            Vector3 thumbTip = activeHand.Fingers[(int)Finger.FingerType.THUMB].TipPosition;
            Vector3 indexTip = activeHand.Fingers[(int)Finger.FingerType.INDEX].TipPosition;
            float actualPinchDist = Vector3.Distance(thumbTip, indexTip);

            if (actualPinchDist >= pinchExitThreshold)
            {
                DropLiver();
                return;
            }

            // KINEMATIC UPDATE
            if (allowTranslation)
                solidLiver.position = Vector3.Lerp(thumbTip, indexTip, 0.5f) + positionOffset;

            if (allowRotation)
                solidLiver.rotation = activeHand.Rotation * rotationOffset;
        }

        public void DropLiver()
        {
            IsGrabbing = false;
            activeHandId = -1;
            trackingLossTimer = 0f;
            
            if (statusText != null) { statusText.text = "SYSTEM IDLE"; statusText.color = Color.white; }
        }
    }
}
```

```csharp
            if (hologramRenderer != null) hologramRenderer.material = defaultHoloMaterial;
            dwellTimer = 0f;
            taskStartTime = Time.time;
            taskActive = true;

            telemetryLogger.StartNewTrialLog(participantID, "HandTracking", currentTask.ToString());
        }

        private void ResetAnatomyPositions()
        {
            Vector3 basePosition = new Vector3(0, 0.871f, 1.172f);
            targetHologram.position = basePosition;
            targetHologram.rotation = Quaternion.identity;

            switch (currentTask)
            {
                case TrialTask.PureTranslation:
                    solidLiver.position = basePosition + new Vector3(-0.3f, 0, 0);
                    solidLiver.rotation = Quaternion.identity;
                    break;
                case TrialTask.PureRotation:
                    solidLiver.position = basePosition;
                    solidLiver.rotation = Quaternion.Euler(Random.Range(45, 135), Random.Range(45, 135), Random.Range(45, 135));
                    break;
                case TrialTask.SixDegreesOfFreedom:
                    solidLiver.position = basePosition + new Vector3(-0.3f, 0.1f, -0.1f);
                    solidLiver.rotation = Quaternion.Euler(Random.Range(45, 135), Random.Range(45, 135), Random.Range(45, 135));
                    break;
            }
        }

        private void ApplyTaskConstraints()
        {
            handManipulator.allowTranslation = (currentTask == TrialTask.PureTranslation || currentTask == TrialTask.SixDegreesOfFreedom);
            handManipulator.allowRotation = (currentTask == TrialTask.PureRotation || currentTask == TrialTask.SixDegreesOfFreedom);
        }

        private void Update()
        {
            if (!taskActive) return;

            float currentDist = Vector3.Distance(solidLiver.position, targetHologram.position);
            float currentRot = Quaternion.Angle(solidLiver.rotation, targetHologram.rotation);

            // Dwell Time Logic
            if (currentDist <= acceptableDistanceError && currentRot <= acceptableAngularError)
            {
                dwellTimer += Time.deltaTime;
                if (statusText != null && !handManipulator.IsGrabbing) 
                    statusText.text = $"ALIGNING... {dwellTimer:F1}s";
                
                if (dwellTimer >= requiredDwellTime)

                    CompleteCurrentTask();
                }
            }
            else
            {
                dwellTimer = 0f; // Reset if user slips out of alignment
            }
        }

        private void CompleteCurrentTask()
        {
            taskActive = false;
            handManipulator.DropLiver();
            handManipulator.enabled = false; // Lock interactions

                statusText.color = Color.yellow;
                statusText.text = $"SUCCESS!\nTask Completion Time (TCT): {tct:F2} seconds";
            }
            
            // Logic to load next scene via SceneManager goes here.
        }
    }
```

```C#
using System.Collections;
using System.Collections.Generic;
using UnityEngine;

namespace VarjoExample
{
    public class Hand : MonoBehaviour
    {
        public Transform xrRig;

        Controller controller;

        public List<Interactable> contactedInteractables = new List<Interactable>();
        private bool triggerDown;
        private FixedJoint fixedJoint = null;
        private Interactable currentInteractable;
        private Rigidbody heldObjectBody;

        void Awake()
        {
            controller = GetComponent<Controller>();
            fixedJoint = GetComponent<FixedJoint>();
        }

        // Update is called once per frame
        void Update()
        {
            if (controller.triggerButton)
            {
                if (!triggerDown)
                {
                    triggerDown = true;
                    Pick();
                }
            }
            else if (!controller.primary2DAxisClick && triggerDown)
            {
                triggerDown = false;
                Drop();
            }
        }

        private void OnTriggerEnter(Collider other)
        {
            if (other.gameObject.CompareTag("Pickable") || other.gameObject.CompareTag("Fracture"))
            {
                contactedInteractables.Add(other.gameObject.GetComponent<Interactable>());
            }
        }

        private void OnTriggerExit(Collider other)
        {
            if (other.gameObject.CompareTag("Pickable") || other.gameObject.CompareTag("Fracture"))
            {
                contactedInteractables.Remove(other.gameObject.GetComponent<Interactable>());
            }
        }

        public void Pick()
        {
            currentInteractable = GetNearestInteractable();

            if (!currentInteractable)
            {
                return;
            }

            // Drop interactable if already held
            if (currentInteractable.activeHand)
            {
                currentInteractable.activeHand.Drop();
            }

            // Attach
            heldObjectBody = currentInteractable.GetComponent<Rigidbody>();
            fixedJoint.connectedBody = heldObjectBody;

            // Set active hand
            currentInteractable.activeHand = this;
        }

        public void Drop()
        {
            if (!currentInteractable)
                return;

            // Detach
            fixedJoint.connectedBody = null;

            // Apply velocity
            heldObjectBody = currentInteractable.GetComponent<Rigidbody>();
            heldObjectBody.velocity = xrRig.TransformVector(controller.DeviceVelocity);
            heldObjectBody.angularVelocity = xrRig.TransformDirection(controller.DeviceAngularVelocity);

            // Clear
            currentInteractable.activeHand = null;
            currentInteractable = null;
        }

        private Interactable GetNearestInteractable()
        {
            Interactable nearest = null;
            float minDistance = float.MaxValue;
            float distance = 0.0f;

            foreach (Interactable interactable in contactedInteractables)
            {
                if (interactable)
                {
```
